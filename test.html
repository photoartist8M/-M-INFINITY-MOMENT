import * as THREE from 'three'; import { startExhibitionSpace } from './exhibitionSpace.js'; // --- renderer --- const canvas = document.getElementById('canvas'); const renderer = new THREE.WebGLRenderer({ canvas, antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); renderer.setPixelRatio(window.devicePixelRatio); // --- camera --- const camera = new THREE.PerspectiveCamera( 75, window.innerWidth / window.innerHeight, 0.1, 1000 ); // exhibitionSpace.js を起動 const exhibition = startExhibitionSpace(renderer, camera); // --- アニメーションループ --- const clock = new THREE.Clock(); function animate() { const dt = clock.getDelta(); exhibition.update(dt); // ← ここで exhibitionSpace.js の update が動く renderer.render(exhibition.scene, camera); requestAnimationFrame(animate); } animate();
